% For leaf, inorder is list of leaf node
% Otherwise compute inorder of L and inorder of R
% append left with Node (N) and Right(R)
inorder(leaf(X),[X]).
inorder(tree(N,L,R),T) :- inorder(L,L1),inorder(R,R1),
    				append(L1,[N|R1],T).

% tests
test_answer :- inorder(tree(1,leaf(2),leaf(3)),T),writeln(T).

test_answer2 :- inorder(tree(a,tree(b,leaf(c),leaf(d)),leaf(e)),T),
    		writeln(T).
